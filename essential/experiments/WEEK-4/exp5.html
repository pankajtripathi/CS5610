In this code I have provided two text boxes for user to enter two numbers. Once user clicks on add,sub,multiply or divide button, appropiate function is called which perfomrs the arithmatic operation

  app = angular.module("controllerDemo", []);
 app.controller("expController",
 function ($scope) {
  $scope.add = function () {
  var num1 = $scope.num1;
  var num2 = $scope.num2;
  $scope.output = parseInt(num1) + parseInt(num2);
 }
 $scope.sub = function () {
  var num1 = $scope.num1;
  var num2 = $scope.num2;
  $scope.output = parseInt(num1) - parseInt(num2);
 }
 $scope.divide = function () {
  var num1 = $scope.num1;
  var num2 = $scope.num2;
  $scope.output = parseInt(num1) / parseInt(num2);
 }
 $scope.multiply = function () {
  var num1 = $scope.num1;
  var num2 = $scope.num2;
  $scope.output = parseInt(num1) * parseInt(num2);
  }
 });
